function sendGPS(lat, lng, accuracy) { fetch(`${BASE_URL}?path=telemetry/gps`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ device_id: DEVICE_ID, ts: new Date().toISOString(), lat, lng, accuracy_m: accuracy }) }); } navigator.geolocation.watchPosition(pos => { const lat = pos.coords.latitude; const lng = pos.coords.longitude; const acc = pos.coords.accuracy; sendGPS(lat, lng, acc); }, err => console.error(err), { enableHighAccuracy: true }); async function loadLatest() { const res = await fetch(`${BASE_URL}?path=telemetry/gps/latest&device_id=${DEVICE_ID}`); const json = await res.json(); if (!json.ok) return; const { lat, lng } = json.data; if (marker) { marker.setLatLng([lat, lng]); } else { marker = L.marker([lat, lng]).addTo(map); } map.setView([lat, lng], 15); } async function loadHistory() { const res = await fetch(`${BASE_URL}?path=telemetry/gps/history&device_id=${DEVICE_ID}&limit=50`); const json = await res.json(); if (!json.ok) return; const points = json.data.items.map(i => [i.lat, i.lng]); if (polyline) { polyline.setLatLngs(points); } else { polyline = L.polyline(points, { color: 'blue' }).addTo(map); } } setInterval(() => { loadLatest(); loadHistory(); }, 5000);
